\subsection*{Real}
\begin{equation}
\int d\ve{v}~e^{-\frac{1}{2}\ve{v}^T\ve{A}\ve{v}}
~=~
(2\pi)^{N/2}~\det\ve{A}^{-1/2}
\end{equation}

\begin{equation}
\int d\ve{v}~e^{-\frac{1}{2}\ve{v}^T\ve{A}\ve{v}+\ve{j}^T\cdot\ve{v}}
~=~
(2\pi)^{N/2}~\det\ve{A}^{-1/2}~e^{\frac{1}{2}\ve{j}^T\ve{A}^{-1}\ve{j}}
\end{equation}

\begin{equation}
\int d\ve{v}~e^{-\frac{1}{2}\ve{v}^T\ve{A}\ve{v}}
~(\ldots)
~=~
(2\pi)^{N/2}~\det\ve{A}^{-1/2}~
\sum\limits_{\text{pairings of}\atop\lbrace i_1,\ldots,i_{2n}\rbrace}A_{i_{k_1}i_{k_2}}^{-1}\ldots A_{i_{k_1}i_{k_2}}^{-1}
\end{equation}
